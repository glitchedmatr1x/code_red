#!/usr/bin/env python3
"""
Code RED camp/vehicle gringo script inspector.

Read-only local inspector for extracted WSC/SCO files. It focuses on the
camp-car lead set:

- playercamp##_gringo.wsc
- campfire##_gringo.wsc
- vehicle_generator.wsc
- car_gringo.wsc
- playercar.wsc
- gen_vehicle_brain.wsc

It extracts printable strings, keyword hit windows, and hashes so the next
step can be evidence-based without modifying archives or game files.

Usage from repo root:
  py -3 tools/codered_inspect_camp_vehicle_gringo_scripts.py --root "C:\\Users\\glitc\\OneDrive\\Desktop\\CodeRED_RPF_Extracts"
"""

from __future__ import annotations

import argparse
import hashlib
import json
import os
import re
from pathlib import Path

TARGET_NAME_PATTERNS = [
    "playercamp", "campfire", "vehicle_generator", "car_gringo", "playercar",
    "carcrank_gringo", "gen_vehicle_brain", "zombie_camp",
]

KEYWORDS = [
    "vehicle", "Vehicle", "VEHICLE", "car", "Car", "CAR", "player", "Player",
    "camp", "Camp", "CAMP", "gringo", "Gringo", "GRINGO", "wsi", "Wsi", "WSI",
    "wagon", "Wagon", "coach", "Coach", "horse", "Horse", "mount", "Mount",
    "script", "Script", "brain", "Brain", "generator", "Generator",
    "ACTOR_VEHICLE_Car01", "Vehicle_Generator", "PlayerCar", "car_gringo",
    "CREATE_ACTOR", "CREATE_ACTOR_IN_LAYOUT", "SET_ACTOR_IN_VEHICLE",
]

SKIP_DIRS = {".git", "__pycache__", ".pytest_cache", "proof_packages", "output"}


def sha1(path: Path) -> str:
    h = hashlib.sha1()
    with path.open("rb") as f:
        for chunk in iter(lambda: f.read(1024 * 1024), b""):
            h.update(chunk)
    return h.hexdigest().upper()


def printable_strings(path: Path, min_len: int = 4) -> list[str]:
    data = path.read_bytes()
    chunks = re.findall(rb"[\x20-\x7E]{%d,}" % min_len, data)
    strings = []
    for c in chunks:
        try:
            s = c.decode("ascii", errors="ignore")
        except Exception:
            continue
        if s.strip():
            strings.append(s)
    return strings


def iter_target_files(root: Path):
    for dirpath, dirnames, filenames in os.walk(root):
        dirnames[:] = [d for d in dirnames if d not in SKIP_DIRS]
        for name in filenames:
            low = name.lower()
            if not low.endswith((".wsc", ".sco", ".xml", ".wgd", ".wsi")):
                continue
            rel = (Path(dirpath) / name).relative_to(root).as_posix().lower()
            if any(pat in rel for pat in TARGET_NAME_PATTERNS):
                yield Path(dirpath) / name


def keyword_windows(strings: list[str], max_windows: int = 30) -> list[dict[str, str]]:
    windows = []
    joined = "\n".join(strings)
    low = joined.lower()
    seen = set()
    for kw in KEYWORDS:
        idx = low.find(kw.lower())
        if idx < 0:
            continue
        start = max(0, idx - 120)
        end = min(len(joined), idx + len(kw) + 180)
        snippet = joined[start:end].replace("\r", " ").replace("\n", " ")
        key = (kw.lower(), snippet[:120])
        if key in seen:
            continue
        seen.add(key)
        windows.append({"keyword": kw, "snippet": snippet[:420]})
        if len(windows) >= max_windows:
            break
    return windows


def classify(rel: str, strings: list[str]) -> list[str]:
    text = (rel + "\n" + "\n".join(strings[:300])).lower()
    tags = []
    if "playercamp" in text:
        tags.append("playercamp_script")
    if "campfire" in text:
        tags.append("campfire_script")
    if "vehicle_generator" in text or "generator" in text and "vehicle" in text:
        tags.append("vehicle_generator_lead")
    if "car_gringo" in text:
        tags.append("car_gringo_lead")
    if "playercar" in text:
        tags.append("playercar_lead")
    if "gen_vehicle_brain" in text or "vehicle_brain" in text:
        tags.append("vehicle_brain_lead")
    if "wsi" in text:
        tags.append("mentions_wsi")
    return tags


def main() -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--root", required=True, help="Extracted RPF root folder")
    ap.add_argument("--out", default="logs/camp_vehicle_gringo_script_inspection", help="Output prefix")
    ap.add_argument("--max-strings", type=int, default=80, help="Max printable strings to keep per file")
    args = ap.parse_args()

    root = Path(args.root).resolve()
    if not root.exists():
        raise SystemExit(f"Missing root: {root}")

    rows = []
    for path in iter_target_files(root):
        rel = path.relative_to(root).as_posix()
        try:
            strings = printable_strings(path)
        except Exception as exc:
            rows.append({
                "relative_path": rel,
                "error": str(exc),
            })
            continue
        hits = sorted({kw for kw in KEYWORDS if any(kw.lower() in s.lower() for s in strings)})
        rows.append({
            "relative_path": rel,
            "size": path.stat().st_size,
            "sha1": sha1(path),
            "tags": classify(rel, strings),
            "keyword_hits": hits,
            "keyword_windows": keyword_windows(strings),
            "printable_string_count": len(strings),
            "sample_strings": strings[: args.max_strings],
        })

    def score(row: dict) -> int:
        tags = row.get("tags", [])
        hits = row.get("keyword_hits", [])
        rel = row.get("relative_path", "").lower()
        value = len(tags) * 20 + len(hits) * 3
        if "playercamp01_gringo.wsc" in rel:
            value += 40
        if "vehicle_generator.wsc" in rel:
            value += 35
        if "car_gringo.wsc" in rel:
            value += 30
        if "playercar.wsc" in rel:
            value += 30
        return value

    rows.sort(key=lambda r: (-score(r), r.get("relative_path", "")))

    out = Path(args.out)
    out.parent.mkdir(parents=True, exist_ok=True)
    json_path = out.with_suffix(".json")
    md_path = out.with_suffix(".md")

    report = {
        "root": str(root),
        "boundary": "Read-only inspection. No archives or game files modified.",
        "file_count": len(rows),
        "rows": rows,
        "next_recommendation": "Compare playercamp01_gringo.wsc with vehicle_generator/car_gringo/playercar before any copied-archive import proof.",
    }
    json_path.write_text(json.dumps(report, indent=2), encoding="utf-8")

    lines = [
        "# Code RED Camp / Vehicle Gringo Script Inspection",
        "",
        f"Root: `{root}`",
        f"Files inspected: `{len(rows)}`",
        "",
        "Boundary: read-only inspection. No archives or game files modified.",
        "",
        "## Priority interpretation",
        "",
        "- `playercamp01_gringo.wsc` is the player-camp behavior lead.",
        "- `vehicle_generator.wsc`, `car_gringo.wsc`, and `playercar.wsc` are vehicle behavior leads.",
        "- Do not replace player camp scripts yet; inspect/correlate first.",
        "",
    ]

    for row in rows[:40]:
        lines.append(f"## `{row.get('relative_path')}`")
        if "error" in row:
            lines.append(f"- error: `{row['error']}`")
            lines.append("")
            continue
        lines.append(f"- size: `{row.get('size')}`")
        lines.append(f"- sha1: `{row.get('sha1')}`")
        lines.append(f"- tags: {', '.join(row.get('tags') or []) or 'none'}")
        lines.append(f"- keyword hits: {', '.join(row.get('keyword_hits') or []) or 'none'}")
        for win in (row.get("keyword_windows") or [])[:5]:
            lines.append(f"- `{win['keyword']}`: {win['snippet']}")
        lines.append("")

    md_path.write_text("\n".join(lines), encoding="utf-8")

    print("# Code RED Camp / Vehicle Gringo Script Inspection")
    print(f"Root: {root}")
    print(f"Files inspected: {len(rows)}")
    print(f"JSON: {json_path}")
    print(f"MD: {md_path}")
    for row in rows[:10]:
        print(f"  {row.get('relative_path')} tags={','.join(row.get('tags') or [])}")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
