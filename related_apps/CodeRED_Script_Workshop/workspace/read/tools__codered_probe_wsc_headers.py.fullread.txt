#!/usr/bin/env python3
"""
Code RED targeted WSC header probe.

Read-only probe for exact .wsc leads. This prints byte headers, ASCII headers,
size, SHA1, and a small string sample for the camp/vehicle gringo WSC files.

Usage from repo root:
  py -3 tools/codered_probe_wsc_headers.py --root "C:\\Users\\glitc\\OneDrive\\Desktop\\CodeRED_RPF_Extracts"

Optional:
  py -3 tools/codered_probe_wsc_headers.py --root "..." --match playercamp01_gringo --match vehicle_generator
"""

from __future__ import annotations

import argparse
import hashlib
import json
import os
import re
from pathlib import Path

DEFAULT_MATCHES = [
    "playercamp01_gringo.wsc",
    "playercamp02_gringo.wsc",
    "playercamp03_gringo.wsc",
    "playercamp04_gringo.wsc",
    "playercampfootlocker.wsc",
    "vehicle_generator.wsc",
    "car_gringo.wsc",
    "carcrank_gringo.wsc",
    "playercar.wsc",
    "gen_vehicle_brain.wsc",
]

SKIP_DIRS = {".git", "__pycache__", ".pytest_cache", "proof_packages", "output"}


def sha1(path: Path) -> str:
    h = hashlib.sha1()
    with path.open("rb") as f:
        for chunk in iter(lambda: f.read(1024 * 1024), b""):
            h.update(chunk)
    return h.hexdigest().upper()


def ascii_preview(data: bytes) -> str:
    return "".join(chr(b) if 32 <= b <= 126 else "." for b in data)


def string_sample(data: bytes, limit: int = 12) -> list[str]:
    chunks = re.findall(rb"[\x20-\x7E]{4,}", data[:512_000])
    out = []
    for chunk in chunks:
        s = chunk.decode("ascii", errors="ignore")[:180]
        if s.strip():
            out.append(s)
        if len(out) >= limit:
            break
    return out


def iter_wsc(root: Path):
    for dirpath, dirnames, filenames in os.walk(root):
        dirnames[:] = [d for d in dirnames if d not in SKIP_DIRS]
        base = Path(dirpath)
        for name in filenames:
            if name.lower().endswith(".wsc"):
                yield base / name


def main() -> int:
    parser = argparse.ArgumentParser()
    parser.add_argument("--root", required=True, help="Extracted RPF root folder")
    parser.add_argument("--out", default="logs/wsc_header_probe", help="Output prefix")
    parser.add_argument("--match", action="append", default=[], help="Filename/path substring to match; repeatable")
    args = parser.parse_args()

    root = Path(args.root).resolve()
    if not root.exists():
        raise SystemExit(f"Missing root: {root}")

    matches = [m.lower() for m in (args.match or DEFAULT_MATCHES)]
    rows = []
    for path in iter_wsc(root):
        rel = path.relative_to(root).as_posix()
        low_rel = rel.lower()
        if not any(m in low_rel for m in matches):
            continue
        data = path.read_bytes()
        rows.append({
            "relative_path": rel,
            "size": len(data),
            "sha1": sha1(path),
            "hex_head_16": data[:16].hex(" ").upper(),
            "hex_head_32": data[:32].hex(" ").upper(),
            "hex_head_64": data[:64].hex(" ").upper(),
            "ascii_head_64": ascii_preview(data[:64]),
            "ascii_head_128": ascii_preview(data[:128]),
            "sample_strings": string_sample(data),
        })

    rows.sort(key=lambda r: r["relative_path"])

    out = Path(args.out)
    out.parent.mkdir(parents=True, exist_ok=True)
    json_path = out.with_suffix(".json")
    md_path = out.with_suffix(".md")

    report = {
        "root": str(root),
        "matches": args.match or DEFAULT_MATCHES,
        "boundary": "Read-only WSC header probe. No files modified.",
        "count": len(rows),
        "rows": rows,
        "interpretation_hint": "Compare WSC hex/ascii heads against extracted SCO heads beginning 53 43 52 02 / SCR. and compiled XSC heads beginning 85 43 53 52 / .CSR.",
    }
    json_path.write_text(json.dumps(report, indent=2), encoding="utf-8")

    lines = [
        "# Code RED WSC Header Probe",
        "",
        f"Root: `{root}`",
        f"Matches: `{len(rows)}`",
        "",
        "Boundary: read-only WSC header probe. No files modified.",
        "",
        "Compare against:",
        "",
        "```text",
        "SCO family: 53 43 52 02 / SCR.",
        "XSC family: 85 43 53 52 / .CSR",
        "```",
        "",
    ]
    for row in rows:
        lines.append(f"## `{row['relative_path']}`")
        lines.append(f"- size: `{row['size']}`")
        lines.append(f"- sha1: `{row['sha1']}`")
        lines.append(f"- hex head 16: `{row['hex_head_16']}`")
        lines.append(f"- hex head 32: `{row['hex_head_32']}`")
        lines.append(f"- ascii head 64: `{row['ascii_head_64']}`")
        if row["sample_strings"]:
            lines.append(f"- strings: `{'; '.join(row['sample_strings'][:5])}`")
        lines.append("")
    if not rows:
        lines.append("No matching WSC files found. Check that the extracted root contains `content/release64/scripting/gringo/commonscripts/`.")
    md_path.write_text("\n".join(lines), encoding="utf-8")

    print("# Code RED WSC Header Probe")
    print(f"Root: {root}")
    print(f"Matches: {len(rows)}")
    print(f"JSON: {json_path}")
    print(f"MD: {md_path}")
    for row in rows:
        print(f"  {row['relative_path']} size={row['size']} head={row['hex_head_16']} ascii={row['ascii_head_64'][:16]}")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
